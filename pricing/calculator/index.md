function final() {

	  var slider1 = document.getElementById('customRange5').value; 

	  if (slider1 <= 35) {
		document.getElementById('totalGrove').value=3300*12; 
		document.getElementById('totalGroveBusiness').value=4100*12; 
	
	  } else if (slider1 > 35) {
		var groveDelta = slider1 - 35;
		
		
		document.getElementById('totalGrove').value=((groveDelta*40)*12)+39600; 
		document.getElementById('totalGroveBusiness').value=((groveDelta*60)*12)+49200; 
	  }
	
	var groveAnnual = parseFloat(document.getElementById('totalGrove').value);
	var groveAnnualBusiness = parseFloat(document.getElementById('totalGroveBusiness').value);
	
	var difference = parseFloat(document.getElementById('totalCosts').value);
	
	var diffTotal = difference - groveAnnual;
	var diffTotalBusiness = difference - groveAnnualBusiness;
	
	document.getElementById('priceDiff').value=parseFloat(diffTotal).toFixed(2);  
	document.getElementById('priceDiffBusiness').value=parseFloat(diffTotalBusiness).toFixed(2);  
	
	
}

}
